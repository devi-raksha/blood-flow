159 inline double
160 get_cell_value(const unsigned int cell_index,
161 const std::string &name,
162 const Vector<double> &data,
163 const std::vector<std::string> &names)
164 {
165 auto it = std::find(names.begin(), names.end(), name);
166 if (it == names.end())
167 return 0.0;
168 const unsigned int col = std::distance(names.begin(), it);
169 // The data vector is stored as [cell0_val0, cell0_val1, ..., cell1_val0,
170 // ...]
171 return data(cell_index * names.size() + col);
172 }

328 template <int dim, int spacedim, typename VectorType>
329 void
330 data_to_dealii_vector(const Triangulation<dim, spacedim> &serial_tria,
331 const Vector<double> &data,
332 const DoFHandler<dim, spacedim> &dh,
333 VectorType &output_vector)
334 {
335 AssertDimension(dh.n_dofs(), output_vector.size());
336 const auto &fe = dh.get_fe();
337
338 const auto dist_to_serial_vertices =
339 distributed_to_serial_vertex_indices(serial_tria, dh.get_triangulation());
340
341 const auto &locally_owned_dofs = dh.locally_owned_dofs();
342
343 types::global_dof_index dofs_offset = 0;
344 unsigned int vertex_comp_offset = 0;
345 unsigned int cell_comp_offset = 0;
346 for (unsigned int field = 0; field < fe.n_blocks(); ++field)
347 {
348 const auto &field_fe = fe.base_element(field);
349 const unsigned int n_comps = field_fe.n_components();
350 if (field_fe.n_dofs_per_vertex() > 0)
351 {
352 // This is a vertex data field
353 const types::global_dof_index n_local_dofs =
354 n_comps * serial_tria.n_vertices();
355 for (const auto &cell : dh.active_cell_iterators())
356 if (cell->is_locally_owned())
357 for (const auto v : cell->vertex_indices())
358 {
359 const types::global_dof_index serial_vertex_index =
360 dist_to_serial_vertices[cell->vertex_index(v)];
361 if (serial_vertex_index != numbers::invalid_unsigned_int)
362 for (unsigned int c = 0; c < n_comps; ++c)
363 {
364 const types::global_dof_index dof_index =
365 cell->vertex_dof_index(v, vertex_comp_offset + c);
366 Assert(locally_owned_dofs.is_element(dof_index),
367 ExcInternalError());
368 output_vector[dof_index] =
369 data[dofs_offset + n_comps * serial_vertex_index +
370 c];
371 }
372 }
373 dofs_offset += n_local_dofs;
374 vertex_comp_offset += n_comps;
375 }
376 else if (field_fe.template n_dofs_per_object<dim>() > 0)
377 {
378 // this is a cell data field
379 const types::global_dof_index n_local_dofs =
380 n_comps * serial_tria.n_global_active_cells();
381
382 // Assumption: serial and parallel meshes have the same ordering of
383 // cells.
384 auto serial_cell = serial_tria.begin_active();
385 auto parallel_cell = dh.begin_active();
386 for (; parallel_cell != dh.end(); ++parallel_cell)
387 if (parallel_cell->is_locally_owned())
388 {
389 // Advanced serial cell until we reach the same cell index of
390 // the parallel cell
391 while (serial_cell->id() < parallel_cell->id())
392 ++serial_cell;
393 const auto serial_cell_index =
394 serial_cell->global_active_cell_index();
395 for (unsigned int c = 0; c < n_comps; ++c)
396 {
397 const types::global_dof_index dof_index =
398 parallel_cell->dof_index(cell_comp_offset + c);
399 Assert(locally_owned_dofs.is_element(dof_index),
400 ExcInternalError());
401 output_vector[dof_index] =
402 data[dofs_offset + n_comps * serial_cell_index + c];
403 }
404 }
405 dofs_offset += n_local_dofs;
406 cell_comp_offset += n_comps;
407 }
408 }
409 }

415 BlockIndices block_indices;
416 for (unsigned int i = 0; i < fe.n_blocks(); ++i)
