54// ---------------------------------------------------------------------------
55// FaceTraceDof
56//
57// One (A_hat, U_hat) trace pair lives on every unique face of the mesh.
58// In 1-D (dim=1, spacedim=3) a "face" is a single vertex / 0-D object,
59// so there is exactly one quadrature point on it and one pair of scalar
60// unknowns per face.
61//
62// face_key = (CellId, local_face_number) — canonical: the *lower* global
63// cell-index side when the face is interior.
64// a_hat_dof = global DOF index of the area trace unknown
65// u_hat_dof = global DOF index of the velocity trace unknown
66// ---------------------------------------------------------------------------
67struct FaceTraceDof
68{
69 types::global_dof_index a_hat_dof;
70 types::global_dof_index u_hat_dof;
71};

415 // Compute the local diastolic cross-sectional area at a cell's centroid
416 // by linearly interpolating in arc-length between r_in (inlet) and
417 // r_out (outlet) of the parent vessel.
418 //
419 // If r_in == r_out == 0 (no taper data), fall back to the vessel-level a_d.
420
421 double
422 compute_a_d_local(
423 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell) const
424 {
425 const unsigned int vid = cell->material_id();
426 const auto &vpp = vessel_map.at(vid);
427
428 if (vpp.r_in <= 0.0 || vpp.r_out <= 0.0)
429 return vpp.a_d;
430
431 const auto &[s_min, s_max] = vessel_s_bounds.at(vid);
432
433 const Tensor<1, spacedim> d_hat = compute_directional_vector(cell);
434
435 const double s = cell->center() * d_hat;
436
437 const double xi_geometry =
438 (s_max - s_min > 1e-14) ? (s - s_min) / (s_max - s_min) : 0.5;
439
440 const double r = vpp.r_in + xi_geometry * (vpp.r_out - vpp.r_in);
441
442 return numbers::PI * r * r;
443 }
444
445 double
446 compute_a_d_at_face(
447 const typename DoFHandler<dim, spacedim>::active_cell_iterator &cell,
448 const unsigned int face_no) const
449 {
450 const unsigned int vid = cell->material_id();
451 const auto &vpp = vessel_map.at(vid);
452 if (vpp.r_in <= 0.0 || vpp.r_out <= 0.0)
453 return vpp.a_d;
454
455 const auto &[s_min, s_max] = vessel_s_bounds.at(vid);
456 const Tensor<1, spacedim> d_hat = compute_directional_vector(cell);
457 const double s = cell->vertex(face_no) * d_hat;
458 const double xi_geometry =
459 (s_max - s_min > 1e-14) ? (s - s_min) / (s_max - s_min) : 0.5;
460 const double r = vpp.r_in + xi_geometry * (vpp.r_out - vpp.r_in);
461 return numbers::PI * r * r;
462 }

465 compute_beta_p(const double E, const double h_wall) const
466 {
467 return (4.0 * std::sqrt(3.14159265358979323846) / 3.0) * E * h_wall;

472
473 double
474 compute_pressure_value(const double A,
475 const unsigned int vid,
476 const double a_d_local) const
477 {
478 const auto &vpp = vessel_map.at(vid);
479 const double beta = compute_beta_p(vpp.E, vpp.h_wall);
480 return vpp.p0 + beta / a_d_local * (std::sqrt(A) - std::sqrt(a_d_local)) +
481 vpp.p_d;
482 }
483
484 double
485 compute_pressure_derivative(const double A,
486 const unsigned int vid,
487 const double a_d_local) const
488 {
489 const auto &vpp = vessel_map.at(vid);
490 const double beta_p = compute_beta_p(vpp.E, vpp.h_wall);
491 return beta_p / (2.0 * a_d_local * std::sqrt(std::max(A, 1e-30)));
492 }

494 double
495 compute_wave_speed(const double A,
496 const unsigned int vid,
497 const double a_d_local) const
498 {
499 const double A_safe = std::max(A, 1e-10);
500 return std::sqrt(A_safe / par["rho"] *

505 compute_wave_speed_derivative(const double A,
506 const unsigned int vid,
507 const double a_d_local) const
508 {
509 const double A_safe = std::max(A, 1e-10);
510 return compute_wave_speed(A_safe, vid, a_d_local) * par["m"] /
