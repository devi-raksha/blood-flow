\section{Governing System}
We consider the one-dimensional blood flow equations parameterized along a centerline tangent vector $\bm b(\bm x)$. The compact vector form of the conservation system is given by:
\begin{equation}
  \partial_t \mathbf{w} + \bm b \cdot \nabla \mathbf{F}(\mathbf{w}) = \mathbf{S}(\mathbf{w}), \qquad
  \mathbf{w} = \begin{pmatrix} A \\ U \end{pmatrix}, \qquad
  \mathbf{S}(\mathbf{w}) = \begin{pmatrix} 0 \\ - \frac{f U}{A} \end{pmatrix},
\end{equation}
where $A$ is the cross-sectional area, $U$ is the mean velocity, $\rho$ is the blood density, and $f$ is the friction coefficient. The non-linear physical flux vector $\mathbf{F}(\mathbf{w})$ is:
\begin{equation}
  \mathbf{F}(\mathbf{w}) = \begin{pmatrix} F_A(\mathbf{w}) \\ F_U(\mathbf{w}) \end{pmatrix} = \begin{pmatrix} A U \\ \frac{U^2}{2} + \frac{1}{\rho} P(A) \end{pmatrix}.
\end{equation}
The arterial wall mechanics are governed by the tube law $$P(A) - P_0 = P_d + \frac{\beta}{A_d} \left[ (A^m - A_d^m \right],$$ with $\beta = 4/3 \sqrt{\pi} E h_w$, yielding the physical wave speed $c(A) = \sqrt{\frac{A}{\rho}\frac{dP}{dA}}$ and , $f = 2(\zeta+2)\mu \pi$.

\section{HDG Variable Spaces}
Let $\Th$ be the collection of non-overlapping line elements $K$, and let $\Eh$ be the set of all boundary endpoints (nodes). We partition $\Eh$ into three mutually disjoint subsets:
\[
  \Eh = \Eho \cup \Ehb \cup \Ehj,
\]
representing standard interior nodes ($\Eho$), exterior boundary terminals ($\Ehb$), and branching junctions ($\Ehj$). 

We define the local element state space $\mathbf{W}_h^k$ and the globally single-valued trace space $\mathbf{M}_h^k$ as:
\begin{align*}
  \mathbf{W}_h^k &= \left\{ \mathbf{v} \in (L^2(\Th))^2 : \mathbf{v}|_K \in (\mathcal{P}^k(K))^2, \, \forall K \in \Th \right\}, \\
  \mathbf{M}_h^k &= \left\{ \bm{\mu} \in (L^2(\Eh))^2 : \bm{\mu}|_F \in (\mathcal{P}^k(F))^2, \, \forall F \in \Eh \right\}.
\end{align*}
the HDG trace variable $\widehat{\mathbf{w}}_h = (\widehat{A}_h, \widehat{U}_h)^\top \in \mathbf{M}_h^k$ is uniquely defined over $\Eh$.

\section{The HDG Weak Formulation}
Inside each element $K \in \Th$, the interior state $\mathbf{w}_h \in \mathbf{W}_h^k$ is driven entirely by the trace variable $\widehat{\mathbf{w}}_h$ through the local weak form:
\begin{equation}
  \left(\partial_t \mathbf{w}_h, \mathbf{v}\right)_K - \left(\mathbf{F}(\mathbf{w}_h), \bm b \cdot \nabla \mathbf{v}\right)_K + \left\langle \widehat{\mathbf{F}}_h \cdot \bm n, \mathbf{v} \right\rangle_{\partial K} = \left(\mathbf{S}(\mathbf{w}_h), \mathbf{v}\right)_K, \quad \forall \mathbf{v} \in (\mathcal{P}^k(K))^2,
  \label{eq:hdg_local}
\end{equation}
where $\bm n$ is the outward unit normal vector pointing out of element $K$ ($\bm n = \pm 1$). 

\subsection{Numerical Trace Flux}
The numerical trace flux vector $\widehat{\mathbf{F}}_h \cdot \bm{n}$ on $\partial K$ is expressed as a unified numerical interface flux function $\mathcal{F}_{\text{num}}$ that depends simultaneously on the internal cell state $\mathbf{w}_h$ and the single-valued global trace $\widehat{\mathbf{w}}_h$:
\begin{equation}
  \widehat{\mathbf{F}}_h \cdot \bm{n} = \mathcal{F}_{\text{num}}(\bm{n}, \mathbf{w}_h, \widehat{\mathbf{w}}_h).
  \label{eq:numerical_flux}
\end{equation}
Depending on the runtime execution context, the operator $\mathcal{F}_{\text{num}}$ evaluates the total interface flux using either a Local Lax-Friedrichs or an HLL discretization closure:
\begin{equation}
  \mathcal{F}_{\text{num}}(\bm{n}, \mathbf{w}_h, \widehat{\mathbf{w}}_h) = \mathbf{F}(\widehat{\mathbf{w}}_h) \cdot \bm{n} + \bm{\Sigma}(\mathbf{w}_h, \widehat{\mathbf{w}}_h)\bigl(\mathbf{w}_h - \widehat{\mathbf{w}}_h\bigr),
\end{equation}
where $\bm{\Sigma}(\mathbf{w}_h, \widehat{\mathbf{w}}_h)$ acts as the local stabilization penalty matrix, taking either the scalar identity form $\alpha_{\text{LF}}\mathbf{I}$ or the directionally-split matrix form $\bm{\Sigma}_{\text{HLL}}$.

\subsection{Unified Global Transmission Condition}

The global HDG problem is completed by requiring that the numerical
trace flux satisfies conservative transmission conditions at all
interfaces. Specifically, we seek
$\widehat{\mathbf{w}}_h \in \mathbf{M}_h^k$ such that
\begin{equation}
  \sum_{K \in \Th}
  \left\langle
  \widehat{\mathbf{F}}_h \cdot \bm n,
  \bm{\mu}
  \right\rangle_{\partial K \setminus (\Ehb \cup \Ehj)}
  +
  \sum_{F \in \Ehb}
  \left\langle
  \widehat{\mathbf{B}}_h,
  \bm{\mu}
  \right\rangle_F
  +
  \sum_{J \in \Ehj}
  \left\langle
  \widehat{\mathbf{J}}_h,
  \bm{\mu}
  \right\rangle_J
  =0,
  \qquad
  \forall \bm{\mu}\in \mathbf{M}_h^k.
\label{eq:global_hdg}
\end{equation}

Here:
\begin{itemize}
\item $\widehat{\mathbf{F}}_h$ denotes the standard HDG interior
      numerical flux,
\item $\widehat{\mathbf{B}}_h$ denotes the physical boundary operator,
\item $\widehat{\mathbf{J}}_h$ denotes the junction coupling operator.
\end{itemize}

\subsubsection{Cell Numerical Flux}

For each element $K\in\Th$, the HDG numerical flux is defined by
\begin{equation}
\widehat{\mathbf F}_h \cdot \bm n
=
\mathbf F(\widehat{\mathbf w}_h)\cdot \bm n
+
\bm{\Sigma}
(\mathbf w_h-\widehat{\mathbf w}_h),
\label{eq:cell_flux}
\end{equation}
where $\bm{\Sigma} = \alpha_{\mathrm{LF}}\mathbf{I}$ is the
Local Lax--Friedrichs stabilization matrix with
\[
\alpha_{\mathrm{LF}}
=
\max\bigl(|U_h \pm c_h|,\,|\widehat{U}_h \pm \widehat{c}_h|\bigr).
\]

\subsubsection{Interior Face Transmission}

Consider an interior face shared by two neighboring elements
$K_L$ and $K_R$ with outward normals satisfying
$\bm n_R = -\bm n_L$.
Since the trace variable $\widehat{\mathbf w}_h$ is single-valued on
the interface, the physical flux contributions cancel:
\[
\mathbf F(\widehat{\mathbf w}_h)\cdot \bm n_L
+
\mathbf F(\widehat{\mathbf w}_h)\cdot \bm n_R
=
0.
\]

Consequently, the interior transmission condition reduces to
\begin{equation}
\mathcal{F}_{\mathrm{int}}(\mathbf{w}_{h,L},\mathbf{w}_{h,R},
\widehat{\mathbf{w}}_h)
=
\bm{\Sigma}_{K_L}(\mathbf w_{h,L}-\widehat{\mathbf w}_h)
+
\bm{\Sigma}_{K_R}(\mathbf w_{h,R}-\widehat{\mathbf w}_h)
=
\mathbf{0}.
\label{eq:interior_transmission}
\end{equation}

The corresponding Jacobian blocks are:
\[
D_f
=
\frac{\partial\mathcal{F}_{\mathrm{int}}}
     {\partial\widehat{\mathbf{w}}_h}
=
-(\bm{\Sigma}_{K_L}+\bm{\Sigma}_{K_R})
=
-2\alpha_{\mathrm{LF}}\mathbf{I},
\]
\[
C_{f,K_{L/R}}
=
\frac{\partial\mathcal{F}_{\mathrm{int}}}
     {\partial\mathbf{w}_{h,{L/R}}}
=
\bm{\Sigma}_{K_{L/R}}
=
\alpha_{\mathrm{LF}}\mathbf{I},
\]
where $\hat{n}_{R}= -\hat{n}_{L}$ 
\subsubsection{Physical Boundary Conditions}

At physical boundaries there is no neighboring element. The trace
$\widehat{\mathbf{w}}_h$ is determined by a transmission residual
$\mathcal{F}_{\mathrm{bc}}(\mathbf{w}_h, \widehat{\mathbf{w}}_h)
= \mathbf{0}$ whose form depends on the boundary type. 

The flux Jacobian matrix and its characteristic splitting are
\[
H(\widehat{\mathbf{w}}_h)
=
\begin{pmatrix}
\widehat{U}_h & \widehat{A}_h \\[4pt]
\dfrac{\widehat{c}_h^2}{\widehat{A}_h} & \widehat{U}_h
\end{pmatrix},
\qquad
H_R = \tfrac{1}{2}(H+|H|),
\qquad
H_L = \tfrac{1}{2}(H-|H|),
\]
where $|H| = R|\Lambda|R^{-1}$ is constructed from the
eigendecomposition of $H$ and
$\widehat{c}_h = c(\widehat{A}_h)$ is the wave speed at the trace.
$H_R \rightarrow $ outgoing characteristics (carrying interior
information outward) and $H_L \rightarrow$ incoming characteristics
(carrying exterior boundary data inward).

\paragraph{Inflow Boundary}

At an inflow face connected to element $K$, the trace
$\widehat{\mathbf{w}}_h = (\widehat{A}_h, \widehat{U}_h)^\top$
is determined by the prescribed flow rate $Q_{\mathrm{in}}(t)$ and
compatibility with the interior through the backward Riemann
invariant $W_2 = U - 4(c - c_0)$:
\begin{equation}
\mathcal{F}_{\mathrm{in}}(\mathbf{w}_h, \widehat{\mathbf{w}}_h)
=
\begin{pmatrix}
\widehat{A}_h\,\widehat{U}_h - Q_{\mathrm{in}}(t)
\\[8pt]
\bigl[\widehat{U}_h - 4(\widehat{c}_h - c_0)\bigr]
-
\bigl[U_h - 4(c_h - c_0)\bigr]
\end{pmatrix}
= \mathbf{0}.
\label{eq:inflow_transmission}
\end{equation}

Row 1 enforces the prescribed flow rate at the trace level. Row 2
enforces continuity of $W_2$ between the trace and the interior,
transmitting information from inside the domain outward.

The Jacobian blocks are:
\begin{equation}
D_{\mathrm{in}}
=
\frac{\partial\mathcal{F}_{\mathrm{in}}}
     {\partial\widehat{\mathbf{w}}_h}
=
\begin{bmatrix}
\widehat{U}_h & \widehat{A}_h
\\[8pt]
4\,\dfrac{dc}{d\widehat{A}_h} & 1
\end{bmatrix},
\qquad
C_{\mathrm{in}}
=
\frac{\partial\mathcal{F}_{\mathrm{in}}}
     {\partial\mathbf{w}_h}
=
\begin{bmatrix}
0 & 0
\\[8pt]
4\,\dfrac{dc}{dA_h} & -1
\end{bmatrix},
B_{\mathrm{in}}
=
\frac{\partial\mathcal{F}_{\mathrm{in}}}
     {\partial\who}
=
\begin{bmatrix}
\hat{U} & \hat{A}
\\[8pt]
-4\hat{c}_h' & 1
\end{bmatrix}.
\label{eq:inflow_jacobian}
\end{equation}

\paragraph{RCR Outlet Boundary}

At terminal outlet boundaries, a three-element Windkessel (RCR)
model represents the downstream vasculature. The capacitor pressure
satisfies
\begin{equation}
C\frac{dP_c}{dt}
=
Q_h - \frac{P_c - P_{\mathrm{out}}}{R_2},
\qquad
Q_h = \widehat{A}_h\,\widehat{U}_h.
\label{eq:RCR_ODE}
\end{equation}

Discretising~\eqref{eq:RCR_ODE} by backward Euler from time level $n$:
\begin{equation}
P_{\mathrm{rcr}}(\widehat{\mathbf{w}}_h)
=
\kappa_0 + \kappa_1\,\widehat{A}_h\,\widehat{U}_h,
\label{eq:Prcr}
\end{equation}
with time-step constants
\begin{equation}
\kappa_0
=
\frac{C\,P_c^n + \dfrac{\Delta t}{R_2}P_{\mathrm{out}}}
     {C + \dfrac{\Delta t}{R_2}},
\qquad
\kappa_1
=
\frac{\Delta t}{C + \dfrac{\Delta t}{R_2}}.
\end{equation}

The total terminal pressure at the outlet face is
\begin{equation}
P_{\infty}(\widehat{\mathbf{w}}_h)
=
P_{\mathrm{rcr}}(\widehat{\mathbf{w}}_h)
+
R_1\,\widehat{A}_h\,\widehat{U}_h
=
\kappa_0 + (\kappa_1 + R_1)\,\widehat{A}_h\,\widehat{U}_h.
\label{eq:Pinf}
\end{equation}

The trace $\widehat{\mathbf{w}}_h$ is determined by pressure
continuity at the outlet face and compatibility with the interior
through the forward Riemann invariant $W_1 = U + 4(c - c_0)$:
\begin{equation}
\mathcal{F}_{\mathrm{rcr}}(\mathbf{w}_h, \widehat{\mathbf{w}}_h)
=
\begin{pmatrix}
P(\widehat{A}_h) - P_{\infty}(\widehat{\mathbf{w}}_h)
\\[8pt]
\bigl[\widehat{U}_h + 4(\widehat{c}_h - c_0)\bigr]
-
\bigl[U_h + 4(c_h - c_0)\bigr]
\end{pmatrix}
= \mathbf{0}.
\label{eq:rcr_transmission}
\end{equation}

Row 1 enforces pressure continuity: the arterial pressure at the
trace must equal the total Windkessel terminal pressure. Row 2
enforces continuity of $W_1$ between the trace and the interior.

The Jacobian blocks are:
\begin{equation}
D_{\mathrm{rcr}}
=
\frac{\partial\mathcal{F}_{\mathrm{rcr}}}
     {\partial\widehat{\mathbf{w}}_h}
=
\begin{bmatrix}
\dfrac{dP}{d\widehat{A}_h}
- (\kappa_1 + R_1)\widehat{U}_h
&
-(\kappa_1 + R_1)\widehat{A}_h
\\[10pt]
4\,\dfrac{dc}{d\widehat{A}_h}
&
1
\end{bmatrix},
\label{eq:Drcr}
\end{equation}

\begin{equation}
C_{\mathrm{rcr}}
=
\frac{\partial\mathcal{F}_{\mathrm{rcr}}}
     {\partial\mathbf{w}_h}
=
\begin{bmatrix}
0 & 0
\\[8pt]
-4\,\dfrac{dc}{dA_h} & -1
\end{bmatrix}.
\label{eq:Brcr}
\end{equation}

\subsubsection{Junction Coupling Conditions}

Consider a junction $J$ connecting $K\ge 3$ vessels. Let
\[
\widehat{\mathbf{w}}_i
=
\begin{pmatrix}
\widehat{A}_i \\ \widehat{U}_i
\end{pmatrix},
\qquad i = 1,\ldots,K,
\]
denote the hybrid trace state of vessel $i$ at the junction, with
orientation sign
\[
s_i =
\begin{cases}
+1, & \text{if vessel } i \text{ ends at the junction},\\
-1, & \text{if vessel } i \text{ starts at the junction}.
\end{cases}
\]

Unlike physical boundaries, a junction is an internal network
interface. No external characteristic information is imposed.
Instead, the standard HDG flux~\eqref{eq:cell_flux} is retained
at each junction face:
\begin{equation}
\widehat{\mathbf{J}}_{h,i}
=
\mathbf{F}(\widehat{\mathbf{w}}_i)\cdot\bm{n}_i
+
\bm{\Sigma}_i(\mathbf{w}_{h,i} - \widehat{\mathbf{w}}_i),
\qquad i = 1,\ldots,K.
\label{eq:junction_flux}
\end{equation}

The junction trace states are determined by the following
$K$ constraints, assembled as the transmission residual:
\begin{equation}
\mathcal{F}_J
(\widehat{\mathbf{w}}_1,\ldots,\widehat{\mathbf{w}}_K)
=
\begin{pmatrix}
\displaystyle\sum_{i=1}^{K}
s_i\,\widehat{A}_i\,\widehat{U}_i
\\[10pt]
\mathcal{H}(\widehat{\mathbf{w}}_1)
- \mathcal{H}(\widehat{\mathbf{w}}_2)
\\[4pt]
\vdots
\\[4pt]
\mathcal{H}(\widehat{\mathbf{w}}_1)
- \mathcal{H}(\widehat{\mathbf{w}}_K)
\end{pmatrix}
= \mathbf{0},
\qquad
\mathcal{H}(\widehat{\mathbf{w}}_i)
=
\tfrac{1}{2}\widehat{U}_i^2 + \Psi(\widehat{A}_i),
\label{eq:junction_transmission}
\end{equation}
where $\Psi(A) = P(A)/\rho$. Row 1 enforces global mass
conservation; rows 2 to $K$ enforce continuity of total energy
head between all vessels and the reference vessel $i=1$.

Since $\mathcal{F}_J$ depends only on junction trace unknowns
and not on any interior element state, the element-to-trace
coupling block vanishes:
\begin{equation}
C_{J,K_i}
=
\frac{\partial\mathcal{F}_J}{\partial\mathbf{w}_{h,i}}
=
\mathbf{0},
\qquad i=1,\ldots,K.
\label{eq:Cjunc}
\end{equation}

The trace control block is the $K\times 2K$ matrix:
\begin{equation}
D_J
=
\frac{\partial\mathcal{F}_J}
     {\partial(\widehat{\mathbf{w}}_1,\ldots,\widehat{\mathbf{w}}_K)}
=
\begin{bmatrix}
s_1\widehat{U}_1 & s_1\widehat{A}_1 &
s_2\widehat{U}_2 & s_2\widehat{A}_2 &
\cdots &
s_K\widehat{U}_K & s_K\widehat{A}_K
\\[8pt]
\Psi'(\widehat{A}_1) & \widehat{U}_1 &
-\Psi'(\widehat{A}_2) & -\widehat{U}_2 &
\cdots & 0 & 0
\\[4pt]
\vdots & & & & \ddots & & \vdots
\\[4pt]
\Psi'(\widehat{A}_1) & \widehat{U}_1 &
0 & 0 & \cdots &
-\Psi'(\widehat{A}_K) & -\widehat{U}_K
\end{bmatrix},
\label{eq:DJ}
\end{equation}
where $\Psi'(\widehat{A}_i) = (1/\rho)\,dP/d\widehat{A}_i$.
The system at the junction is closed by the $B$ blocks from the
element equations, which supply the remaining $K$ equations
through the HDG numerical flux~\eqref{eq:junction_flux}.

\item two daughter vessels $D_1$ and $D_2$,
\item one inflow boundary,
\item one bifurcation junction,
\item two RCR outlet boundaries.
\end{itemize}

\subsection{Global Newton System}

After assembling all transmission residuals, the Newton linearisation
at each time step yields the block system
\begin{equation}
\begin{pmatrix}
A_P    & 0      & 0      & B_P    \\
0      & A_{D_1}& 0      & B_{D_1}\\
0      & 0      & A_{D_2}& B_{D_2}\\
C_P    & C_{D_1}& C_{D_2}& D
\end{pmatrix}
\begin{pmatrix}
\delta W_P \\ \delta W_{D_1} \\ \delta W_{D_2} \\ \delta\widehat{W}
\end{pmatrix}
=
-
\begin{pmatrix}
R_P \\ R_{D_1} \\ R_{D_2} \\ R_{\widehat{W}}
\end{pmatrix}.
\label{eq:block_hdg_system}
\end{equation}

The trace unknown vector, ordered by face type, is
\begin{equation}
\widehat{W}
=
\begin{pmatrix}
\widehat{W}_{\mathrm{in}}   \\
\widehat{W}_{P,1}           \\
\widehat{W}_{P,2}           \\
\widehat{W}_{J}             \\
\widehat{W}_{D_1,1}         \\
\widehat{W}_{D_1,2}         \\
\widehat{W}_{\mathrm{out},1}\\
\widehat{W}_{D_2,1}         \\
\widehat{W}_{D_2,2}         \\
\widehat{W}_{\mathrm{out},2}
\end{pmatrix},
\end{equation}
where each entry is a 2-vector $(\widehat{A}, \widehat{U})^\top$
at the corresponding face, where \begin{align}
A_\alpha 
&= \frac{\partial R_\alpha}{\partial W_\alpha}
& &\longrightarrow \text{element residual w.r.t. element unknowns}
\\[6pt]
B_\alpha 
&= \frac{\partial R_\alpha}{\partial \widehat{W}}
& &\longrightarrow \text{element residual w.r.t. trace unknowns}
\\[6pt]
C_\alpha 
&= \frac{\partial R_{\widehat{W}}}{\partial W_\alpha}
& &\longrightarrow \text{transmission residual w.r.t. element unknowns}
\\[6pt]
D 
&= \frac{\partial R_{\widehat{W}}}{\partial \widehat{W}}
& &\longrightarrow \text{transmission residual w.r.t. trace unknowns}
\end{align}

\subsection{Element Residual Vectors}

The vessel residual vectors $R_\alpha$ appearing in the right-hand
side of~\eqref{eq:block_hdg_system} are assembled from the local
weak form residual on each element $K \in \alpha$:

\begin{equation}
R_\alpha
=
\begin{pmatrix}
R_{K_1} \\ R_{K_2} \\ R_{K_3}
\end{pmatrix},
\end{equation}

where for each element $K$ the local residual is:
\begin{equation}
R_K(\mathbf{w}_h, \widehat{\mathbf{w}}_h)
=
\underbrace{
\left(\frac{\mathbf{w}_h - \mathbf{w}_h^n}{\Delta t},
\mathbf{v}\right)_K
}_{\text{time derivative}}
-
\underbrace{
\left(\mathbf{F}(\mathbf{w}_h),
\bm{b}\cdot\nabla\mathbf{v}\right)_K
}_{\text{volume flux}}
+
\underbrace{
\left\langle
\widehat{\mathbf{F}}_h \cdot \bm{n},
\mathbf{v}
\right\rangle_{\partial K}
}_{\text{face flux via trace}}
-
\underbrace{
\left(\mathbf{S}(\mathbf{w}_h),
\mathbf{v}\right)_K
}_{\text{source}},
\label{eq:RK}
\end{equation}

with the numerical flux $\widehat{\mathbf{F}}_h$ defined
in~\eqref{eq:cell_flux}. The face flux term couples $R_K$
to the trace unknowns $\widehat{\mathbf{w}}_h$, which is
why the $B_\alpha$ blocks appear in the element rows of
the global system.

Similarly, $R_{\widehat{W}}$ is the vector of transmission
residuals assembled from all faces:
\begin{equation}
R_{\widehat{W}}
=
\begin{pmatrix}
\mathcal{F}_{\mathrm{in}}(\mathbf{w}_h, \widehat{W}_{\mathrm{in}})   \\
\mathcal{F}_{f}(\mathbf{w}_h, \widehat{W}_{P,1})                     \\
\mathcal{F}_{f}(\mathbf{w}_h, \widehat{W}_{P,2})                     \\
\mathcal{F}_{J}(\widehat{W}_{J})                                      \\
\mathcal{F}_{f}(\mathbf{w}_h, \widehat{W}_{D_1,1})                   \\
\mathcal{F}_{f}(\mathbf{w}_h, \widehat{W}_{D_1,2})                   \\
\mathcal{F}_{\mathrm{rcr}}(\mathbf{w}_h, \widehat{W}_{\mathrm{out},1})\\
\mathcal{F}_{f}(\mathbf{w}_h, \widehat{W}_{D_2,1})                   \\
\mathcal{F}_{f}(\mathbf{w}_h, \widehat{W}_{D_2,2})                   \\
\mathcal{F}_{\mathrm{rcr}}(\mathbf{w}_h, \widehat{W}_{\mathrm{out},2})
\end{pmatrix},
\label{eq:R_What}
\end{equation}

each entry being the transmission residual at the corresponding
face as defined in
equations~\eqref{eq:interior_transmission},
\eqref{eq:inflow_transmission},
\eqref{eq:rcr_transmission}, and
\eqref{eq:junction_transmission}.

\subsection{Structure of the Trace Block $D$}

The trace block $D$ is block-diagonal, with each diagonal entry
being the Jacobian $D_F = \partial\mathcal{F}_F/\partial
\widehat{\mathbf{w}}_h$ at the corresponding face:

\begin{equation}
D
=
\mathrm{blockdiag}
\bigl(
D_{\mathrm{in}},\;
D_{f},\;
D_{f},\;
D_{J},\;
D_{f},\;
D_{f},\;
D_{\mathrm{rcr}},\;
D_{f},\;
D_{f},\;
D_{\mathrm{rcr}}
\bigr),
\label{eq:D_blockdiag}
\end{equation}
with the $2\times2$ values:
\begin{equation}
\renewcommand{\arraystretch}{1.8}
\begin{array}{ll}
D_{\mathrm{in}}
=
\begin{bmatrix}
\widehat{U}_h & \widehat{A}_h \\
4\,dc/d\widehat{A}_h & 1
\end{bmatrix},
&
\text{(inflow face)}
\\[10pt]
D_{f}
=
-2\alpha_{\mathrm{LF}}\,\mathbf{I},
&
\text{(interior face)}
\\[10pt]
D_{\mathrm{rcr}}
=
\begin{bmatrix}
dP/d\widehat{A}_h - (\kappa_1{+}R_1)\widehat{U}_h
& -(\kappa_1{+}R_1)\widehat{A}_h \\
4\,dc/d\widehat{A}_h & 1
\end{bmatrix},
&
\text{(RCR outlet face)}
\end{array}
\label{eq:D_blocks_summary}
\end{equation}
and $D_J$ is the $3\times6$ block from equation~\eqref{eq:DJ},
spanning the three junction trace unknowns
$(\widehat{W}_{J,P},\, \widehat{W}_{J,D_1},\, \widehat{W}_{J,D_2})$.
