\section{Model problem and geometry}

We consider the one-dimensional blood flow equations embedded along a
centerline $\Gamma\subset\R^{\text{spacedim}}$, parameterized by a unit tangent
vector $\bm b(\bm x)$:
\begin{align}
  \partial_t A + \nabla \cdot (\bm b A U) &= 0,
  && \text{(Area equation)}, \label{eq:area-eq}\\[2mm]
  \partial_t U + \nabla \cdot \bm b  \!\left(\frac{U^2}{2}  + \frac{1}{\rho} P(A)\right)
  + c\,U &= 0,
  && \text{(Momentum equation).} \label{eq:momentum-eq}
\end{align}
Here $A$ and $U$ denote the cross-sectional area and mean velocity,
respectively.
The direction vector of each 1D element is
\[
  \bm b = \frac{\bm x_1 - \bm x_0}{\lVert \bm x_1 - \bm x_0\rVert},
  \qquad
  \beta := \bm b\!\cdot\!\bm n
\]
where $\bm n$ is the outward normal. The pressure $P(A)$ is governed by a
constitutive \emph{tube law}, for example
\[
  P(A) = P_0 + \mu \!\left[\!\left(\frac{A}{A_0}\right)^m - 1\!\right],
  \qquad
  \frac{dP}{dA} = \mu \,m\,\frac{A^{m-1}}{A_0^m},
\]
and the wave speed is
\[
  c = \sqrt{\frac{A}{\rho}\frac{dP}{dA}}.
\]

The compact vector form of the system is
\begin{equation}
  \partial_t \mathbf{W} + \nabla \cdot (\bm b \mathbf{F}(\mathbf{W}))
  = S(\mathbf{W}),
  \qquad
  \mathbf{W} =
  \begin{pmatrix} A \\[1mm] U \end{pmatrix},
  \quad
  S(\mathbf{W}) = -
  \begin{pmatrix} 0 \\[1mm] cU \end{pmatrix},
  \label{eq:model-vector}
\end{equation}
with the physical flux
\[
  \mathbf{F}(\mathbf{W}) =
  \begin{pmatrix}
    A U \\[2mm]
    \dfrac{U^2}{2} + \dfrac{1}{\rho} P(A)
  \end{pmatrix}.
\]


\section{Discontinuous Galerkin weak formulation}


Let $\Th$ denote a partition of $\Gamma$ into 1D elements $K$, and for
polynomial degree $k\ge0$ define the discontinuous finite element space
\[
  \Vh = \{v\in L^2(\Gamma): v|_K \in \mathbb P_k(K),\ \forall K\in\Th\}.
\]
For an interior face $F\in\Fh^\circ$ across two elements $E^+$ and $E^-$, with
outward unit normals $\bm n^+$ and $\bm n^-$, the jump and average operators are
\[
  \jump{v}=v^-\bm n^- \cdot \bm b^- + v^+\bm n^- \cdot \bm b^+,
  \qquad
  \avg{v}=\tfrac12(v^-+v^+),
\]
and on boundary faces $F\subset\partial\Gamma$ we set
$\avg{v}=v$ and $\jump{v}=v\bm n$

DG weak form for \eqref{eq:model-vector} is 
\begin{equation}{\label{semi-discrete-weak-form}}
     \left( \partial_t \mathbf{W}_h , \phi \right)
  - (\mathbf{F}(\mathbf{W}), \bm b\cdot \nabla \phi) + \langle \widehat{\mathbf{F}}(\mathbf{W}), \jump{\phi} \rangle
  - (S(\mathbf{W}_h), \phi) = 0,
\end{equation}
where $\widehat{\mathbf{F}}$ is numerical flux defined later.

\section{Time integration with ARKode}

The deal.II wrapper \texttt{SUNDIALS::ARKode} expects the semi-discrete system
in the form
\begin{equation}\label{eq:ark-ivp}
  \mathcal{M}\,\dot{\mathbf{y}}
  = f_E(t,\mathbf{y}) + f_I(t,\mathbf{y}),
  \qquad \mathbf{y}\equiv\mathbf{W}_h\in\Vh^2,
\end{equation}
where $\mathcal{M}$ is a (possibly time-dependent) mass operator acting on
vectors in $\Vh^2$, and $f_E$ and $f_I$ denote the explicit and implicit
contributions to the right-hand side.  Setting either $f_E\equiv0$ or
$f_I\equiv0$ recovers the explicit RK or DIRK limits supported by ARKode.

\subsection*{DG operators in ARKode notation}

For any pair of test functions $\bm\phi=(\phi_A,\phi_U)\in\Vh^2$, the DG mass
operator is
\[
  (\mathcal{M}\,\mathbf{v},\bm\phi)
  := (\mathbf{v},\bm\phi)
  = \sum_{K\in\Th} \int_K \mathbf{v}\cdot\bm\phi\,dx .
\]
We collect all spatial terms (physical flux, numerical flux, source) in the
bilinear form
\[
  \mathcal{L}(\mathbf{W}_h;\bm\phi)
  :=
      (\mathbf{F}(\mathbf{W}_h),\bm b\!\cdot\!\nabla\bm\phi)
      - \langle \widehat{\mathbf{F}}(\mathbf{W}_h),\jump{\bm\phi} \rangle
      + (S(\mathbf{W}_h),\bm\phi),
\]
so that \eqref{semi-discrete-weak-form} reads
$(\mathcal{M}\,\dot{\mathbf{W}}_h,\bm\phi) = \mathcal{L}(\mathbf{W}_h;\bm\phi)$.
The ARKode callbacks are then
\[
  (f_E(t,\mathbf{W}_h),\bm\phi) = \mathcal{L}_E(\mathbf{W}_h;\bm\phi),\qquad
  (f_I(t,\mathbf{W}_h),\bm\phi) = \mathcal{L}_I(\mathbf{W}_h;\bm\phi),
\]
with $\mathcal{L}_E+\mathcal{L}_I=\mathcal{L}$.  In practice we use the fully
implicit choice $\mathcal{L}_E\equiv0$ and $\mathcal{L}_I\equiv\mathcal{L}$,
which mirrors the backward Euler treatment used before; other ImEx splits
(e.g. explicit advection, implicit damping) can be obtained by moving terms
between $\mathcal{L}_E$ and $\mathcal{L}_I$ without changing the formulas
below.

\subsection*{Implicit stages and Newton system}

For an implicit ARK stage $i$ with step size $h_n$ and Butcher coefficient
$\gamma=A^I_{i,i}$, ARKode solves the nonlinear stage equation
\begin{equation}\label{eq:ark-stage}
  G(z_i) := \mathcal{M}\,z_i
    - h_n\,\gamma\,f_I(t^I_{n,i},z_i) - a_i = 0,
\end{equation}
where $a_i$ collects already-known information from previous stages and the
explicit part,
\[
  a_i
  = \mathcal{M}\,y_{n-1}
    + h_n \sum_{j=1}^{i-1}
      \Big(
        A^E_{i,j}\,f_E(t^E_{n,j},z_j)
        + A^I_{i,j}\,f_I(t^I_{n,j},z_j)
      \Big).
\]
When $f_I$ is nonlinear, ARKode uses Newton's method
$z_i^{m+1}=z_i^{m}+\delta z^{m+1}$ with linear systems
\begin{equation}\label{eq:ark-newton}
  N(z_i^{m})\,\delta z^{m+1} = -G(z_i^{m}),
  \qquad
  N(z) := \mathcal{M} - \gamma\,J(z),
\end{equation}
where $J(z) = (\partial f_I / \partial y) (z)$ is the Jacobian of $f_I$
evaluated in $z$.

\subsection*{Jacobian action for the implicit operator}

Using the same notation as in \eqref{semi-discrete-weak-form}, the weak action
of $J(\mathbf{W}_h)$ on a perturbation $\delta\mathbf{W}_h$ is
\begin{equation}\label{eq:ark-jacobian}
\boxed{
\begin{aligned}
  \bigl(J(\mathbf{W}_h)\,\delta\mathbf{W}_h,\bm\phi\bigr)
  &=
    (\mathbf{H}(\mathbf{W}_h)\,\delta\mathbf{W}_h,\bm b\!\cdot\!\nabla\bm\phi)
    - \langle \delta\widehat{\mathbf{F}}(\mathbf{W}_h;\delta\mathbf{W}_h),
              \jump{\bm\phi} \rangle \\
  &\qquad\qquad
    + \big(\tfrac{\partial S}{\partial\mathbf{W}}(\mathbf{W}_h)\,\delta\mathbf{W}_h,
       \bm\phi\big),
\end{aligned}}
\end{equation}
with the physical flux Jacobian
\[
  \mathbf{H}(\mathbf{W})
  =
  \begin{pmatrix}
    U & A \\[1mm]
    \dfrac{c^2}{A} & U
  \end{pmatrix},
  \qquad
  c = \sqrt{\frac{A}{\rho}\frac{dP}{dA}(A)}.
\]
For the Lax--Friedrichs numerical flux the consistent linearization appearing
in \eqref{eq:ark-jacobian} is
\begin{equation}\label{eq:LF-derivative-ark}
  \delta\widehat{\mathbf{F}}
  =
  \tfrac12\Big(
    \mathbf{H}^-\,\delta\mathbf{W}^-\,\beta^-
    + \mathbf{H}^+\,\delta\mathbf{W}^+\,\beta^+
  \Big)
  - \tfrac12\,\alpha
    (\delta\mathbf{W}^+ - \delta\mathbf{W}^-),
  \qquad \beta^\pm := \bm b\!\cdot\!\bm n^\pm ,
\end{equation}
where $\alpha$ is the usual Rusanov penalty
$\alpha=\max(|U^\pm\!-\!c^\pm|,|U^\pm\!+\!c^\pm|)$.  Componentwise,
\[
\begin{aligned}
\delta\widehat{F}_A
&= \tfrac12\Big[(U^-\,\delta A^- + A^-\,\delta U^-)\beta^-
                 + (U^+\,\delta A^+ + A^+\,\delta U^+)\beta^+\Big]
   - \tfrac12\alpha(\delta A^+ - \delta A^-), \\[1mm]
\delta\widehat{F}_U
&= \tfrac12\Big[\big(\tfrac{(c^-)^2}{A^-}\delta A^- + U^-\,\delta U^-\big)\beta^-
                 + \big(\tfrac{(c^+)^2}{A^+}\delta A^+ + U^+\,\delta U^+\big)\beta^+\Big]
   - \tfrac12\alpha(\delta U^+ - \delta U^-).
\end{aligned}
\]
Inserting \eqref{eq:ark-jacobian} in \eqref{eq:ark-newton} yields the matrix-free
operator needed by \texttt{jacobian\_times\_vector} in
\texttt{deal.II/sundials/arkode.h}; the mass term in \eqref{eq:ark-newton} is
handled by the ARKode wrapper through the supplied $\mathcal{M}$.

\subsection*{Boundary data inside the ARKode callbacks}

When evaluating either $f_I$ or $J$, the exterior trace $\mathbf{W}^+$ in the
flux and its variation follows the same characteristic-based logic as before.
Let $\lambda_{1,2}=U\mp c$ denote the characteristic speeds.  The mapping from
interior data to boundary data is summarised in
Table~\ref{tab:bd-arkode}; it is applied identically inside
$\widehat{\mathbf{F}}$ and $\delta\widehat{\mathbf{F}}$ in
\eqref{eq:LF-derivative-ark}.

\begin{remark}
\begin{itemize}
  \item The choice $f_E\equiv0$ yields a DIRK scheme identical in spirit to the
  backward Euler step used previously; switching terms between $f_E$ and $f_I$
  gives ImEx schemes without altering the formulas for $\widehat{\mathbf{F}}$
  and $J$.
  \item In code, \texttt{explicit\_function} and \texttt{implicit\_function}
  assemble the vectors corresponding to $\mathcal{L}_E$ and $\mathcal{L}_I$,
  \texttt{mass\_operator} (or the default mass matrix action) supplies
  $\mathcal{M}$, and \texttt{jacobian\_times\_vector} applies
  $N(\cdot)=\mathcal{M}-\gamma J(\cdot)$ using \eqref{eq:ark-jacobian}.
  \item Newton updates returned by ARKode are coefficient vectors; recovering
  the physical variables uses $\mathbf{W}_h^{m+1}=z_i^{m}+\delta z^{m+1}$ at the
  stage level and $\mathbf{W}_h^{n+1}=y_n$ at the time-step level, matching the
  notation of \eqref{eq:ark-ivp}.
\end{itemize}
\end{remark}
